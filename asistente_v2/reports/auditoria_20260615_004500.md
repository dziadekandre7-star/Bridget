import json
import os

class Memoria:
    def __init__(self, archivo='memory.json'):
        self.archivo = archivo
        self.disparadores = [
            "me puedes decir",
            "me podes decir",
            "puedes decirme",
            "podes decirme",
            "llamame ",
            "llámame ",
        ]

    def cargar_recuerdos(self):
        try:
            with open(self.archivo, 'r', encoding='utf-8') as archivo:
                return json.load(archivo)
        except (json.JSONDecodeError, FileNotFoundError):
            return []

    def guardar_recuerdo(self, recuerdo):
        if not isinstance(recuerdo, str):
            raise ValueError("El recuerdo debe ser una cadena")

        recuerdos = self.cargar_recuerdos()
        if recuerdo not in recuerdos:
            recuerdos.append(recuerdo)
            try:
                with open(self.archivo, 'w', encoding='utf-8') as archivo:
                    json.dump(recuerdos, archivo, ensure_ascii=False, indent=4)
                return True
            except Exception as e:
                print(f"Error al guardar recuerdo: {e}")
        return False

    def leer_recuerdos(self):
        return self.cargar_recuerdos()

    def olvidar_recuerdo(self, recuerdo):
        if not isinstance(recuerdo, str):
            raise ValueError("El recuerdo debe ser una cadena")

        recuerdos = self.cargar_recuerdos()
        if recuerdo in recuerdos:
            recuerdos.remove(recuerdo)
            try:
                with open(self.archivo, 'w', encoding='utf-8') as archivo:
                    json.dump(recuerdos, archivo, ensure_ascii=False, indent=4)
                return True
            except Exception as e:
                print(f"Error al olvidar recuerdo: {e}")
        return False

    def borrar_todos_los_recuerdos(self):
        try:
            with open(self.archivo, 'w', encoding='utf-8') as archivo:
                json.dump([], archivo, ensure_ascii=False, indent=4)
            return True
        except Exception as e:
            print(f"Error al borrar todos los recuerdos: {e}")
            return False

    def obtener_nombre_preferido(self):
        recuerdos = self.cargar_recuerdos()
        for recuerdo in recuerdos:
            recuerdo_normalizado = recuerdo.lower()
            for disparador in self.disparadores:
