import whisper
import sounddevice as sd
import numpy as np
import webrtcvad
from scipy.io.wavfile import write
import tempfile
import logging

# Configuración de la ruta de archivo temporal
temp_dir = tempfile.gettempdir()

# Configuración del modelo de reconocimiento de voz
MODELO_WHISPER = whisper.load_model("base", device="cpu")

# Configuración de la frecuencia y duración de silencio
FRECUENCIA = 16000
DURACION_SILENCIO = 1.6  # segundos de silencio para cortar

def escuchar():
    """
    Función para escuchar y transcribir audio a texto
    """
    logging.info("Escuchando...")
    vad = webrtcvad.Vad(2)  # Agresividad 0-3

    frames = []
    silencio_frames = 0
    hablando = False
    frames_por_segundo = 50
    frame_duracion = 1000 // frames_por_segundo  # ms por frame
    frame_size = FRECUENCIA * frame_duracion // 1000

    with sd.InputStream(samplerate=FRECUENCIA, channels=1, dtype='int16') as stream:
        while True:
            frame, _ = stream.read(frame_size)
            frame_bytes = frame.tobytes()

            es_voz = vad.is_speech(frame_bytes, FRECUENCIA)

            if es_voz:
                hablando = True
                silencio_frames = 0
                frames.append(frame)
            elif hablando:
                silencio_frames += 1
                frames.append(frame)

                if silencio_frames > frames_por_segundo * DURACION_SILENCIO:
                    break

    audio = np.concatenate(frames, axis=0)
    temp_file = tempfile.NamedTemporaryFile(suffix=".wav", dir=temp_dir)
    write(temp_file.name, FRECUENCIA, audio)
    resultado = MODELO_WHISPER.transcribe(temp_file.name, language="es")
    temp_file.close()
    return resultado["text"].strip()

def escuchar_audio(ruta_archivo):
    """
    Función para transcribir audio a texto desde un archivo
    """
    try:
        resultado = MODELO_WHISPER.transcribe(ruta_archivo, language="es")
        return resultado["text"].strip()
    except whisper.Error as e:
        logging.error(f"Error al transcribir: {e}")
        return None

def escuchar_fragmento(duracion=1.5):
    """
    Función para escuchar y transcribir un fragmento de audio
    """
    audio = sd.rec(int(duracion * FRECUENCIA), samplerate=FRECUENCIA, channels=1, dtype='int16')
    sd.wait()
    temp_file = tempfile.NamedTemporaryFile(suffix=".wav", dir=temp_dir)
    write(temp_file.name, FRECUENCIA, audio)
    resultado = MODELO_WHISPER.transcribe(temp_file.name, language="es")
    temp_file.close()
    return resultado["text"].strip().lower()

if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO)
    escuchar()
